	desc = "EVTDESC954957"
	picture = "Rebellion"

	    in_sphere = PRU
		primary_culture = north_german
		NOT = {

		NOT = {
			has_country_flag = prussian_influence_present
		}
	}
	
	mean_time_to_happen = {
		days = 1

		}

			factor = 1.0
		}
	}
}
